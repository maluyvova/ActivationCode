33      public static void androidWebChrome() throws MalformedURLException, InterruptedException {
34          String code=WebFireTVGetCode.getCode("Firefox");
35          AndroidDriver<AndroidElement> driver=capabilities();
36          Actions action =new Actions(driver);
37          WebDriverWait wait = new WebDriverWait (driver,500);
38          driver.get(URLS.webUrl);
39          WebElement signInButton=wait.until(ExpectedConditions.presenceOfElementLocated(By.xpath(XpathForWebActivationSgnInScreen.signInButton)));
40          action.click(signInButton).build().perform();
41          //action.click(waits(XpathForWebActivationSgnInScreen.signInButton)).click();
42          WebElement emailField= wait.until(ExpectedConditions.presenceOfElementLocated(By.xpath(XpathForWebActivationSgnInScreen.emailField)));
43          action.click(emailField).build().perform();
44          action.sendKeys(emailField, "vburian@tubi.tv").build().perform();
45          WebElement passwordlField= wait.until(ExpectedConditions.presenceOfElementLocated(By.xpath(XpathForWebActivationSgnInScreen.passwordFiled)));
46          action.click(passwordlField).build().perform();
47          action.sendKeys(passwordlField, "tubitv").build().perform();
48          WebElement sumbitButtonForSignIn= wait.until(ExpectedConditions.presenceOfElementLocated(By.xpath(XpathForWebActivationSgnInScreen.submitButoonForSignIn)));
49          action.click(sumbitButtonForSignIn).build().perform();
50          WebElement activationCodeField= wait.until(ExpectedConditions.presenceOfElementLocated(By.xpath(XpathForWebActivationSgnInScreen.activationCodeField)));
51          action.moveToElement(activationCodeField).build().perform();
52          action.click(activationCodeField).build().perform();
53          action.sendKeys(activationCodeField,code).build().perform();
54          WebElement submitForActivationCodeField= wait.until(ExpectedConditions.presenceOfElementLocated(By.xpath(XpathForWebActivationSgnInScreen.submitButtonForActivationCode)));
55          action.click(submitForActivationCodeField).build().perform();
56          WebElement homeButtonAfterActivationDevice= wait.until(ExpectedConditions.presenceOfElementLocated(By.xpath(XpathForWebActivationSgnInScreen.homebuttonAfterActivationDevice)));
57          action.click(homeButtonAfterActivationDevice).build().perform();
58          ArrayList<WebElement> containerWithFeaturedTitles= (ArrayList<WebElement>) wait.until(ExpectedConditions.presenceOfAllElementsLocatedBy(By.xpath(XpathForWebActivationSgnInScreen.containerWithFeaturetTitles)));
59          int size=containerWithFeaturedTitles.size();
60          boolean ContainerFeaturedTitlesInFireTV=FireTVScreen.isContainerOfFeaturedTitlesDisplayed();
61          Assert.assertTrue(ContainerFeaturedTitlesInFireTV, "Featured titles is not displayed in FireTVScreen after activation in AndroidChrome");
62          Assert.assertEquals(1,size);
63      }
